# Stack

```swift
class myStack {
     var stack = [Int]()
    
    func isFull() -> Bool {
        if stack.isEmpty {
            return false
        } else {
            return true
        }
    }
    
    func push(_ value: Int){
        stack.append(value)
    }
    
    func pop() -> Int {
        if stack.isEmpty {
            return -1
        } else {
            var tmp = stack[0]
            stack.removeFirst()
            return tmp
        }
    }
    
    func myPrint(){
        for n in stack{
            print(n, terminator: " ")
        }
    }
}

print("첫번째 스택 확인")
let s: myStack = myStack()
print(s.isFull()) // false
s.push(10)
print(s.isFull()) // true
s.push(5)
s.push(3)
s.push(7)
print(s.pop()) // 10 
s.myPrint() // 5 3 7

print("\n\n두번째 스택 확인")
let s2: myStack = myStack()
print(s2.isFull()) // false
s2.push(3)
s2.push(6)
s2.push(9)
print(s2.pop()) // 3
s2.myPrint() // 6 9
```

# Queue

```swift

class Queue {
    var queue = Array<Int>()
    
    func isFull() -> Bool {
        if queue.isEmpty{
            return false
        } else {
            return true
        }
    }
    
    func enQueue(_ value: Int) {
        queue.append(value)
    }
    
    func deQueue() -> Int {
        if queue.isEmpty {
            return -1
        } else {
            var i = queue.count - 1
            var tmp = queue[i]
            queue.removeLast()
            return tmp
        }
    }
    
    func myPrint() {
        for e in queue{
            print(e, terminator: "->")
        }
    }
}

var q: Queue = Queue()
print(q.isFull()) // false
q.enQueue(1)
q.enQueue(2)
q.enQueue(3)
q.enQueue(4)
print(q.deQueue()) // 4
q.myPrint() // 1->2->3->
```
